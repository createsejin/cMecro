String GetString = "";

void setup() {
  Serial.begin(9600);

}

void loop() {
  GetString = "";
  char data = -1;
  while (Serial.available()) {
    data = Serial.read();
    GetString.concat(data);
  }

  if (GetString != "") {
    String sub1 = GetString.substring(0, 2);
    int sub2 = GetString.substring(2).toInt();
    Serial.println(sub1+sub2);
    delay(100);
  }
}
